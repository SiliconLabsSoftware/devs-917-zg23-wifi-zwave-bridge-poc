167/* RSSI feedback constants */
168/* This is a signed 8-bit value. Note that values from RSSI_RESERVED_START to
169 * 124 are reserved. All values below RSSI_RESERVED_START are received power
170 * in dBms. Other values are defined below. */
171#define RSSI_NOT_AVAILABLE 127 /* RSSI measurement not available */
172#define RSSI_MAX_POWER_SATURATED 126 /* Receiver saturated. RSSI too high to measure precisely. */
173#define RSSI_BELOW_SENSITIVITY 125 /* No signal detected. The RSSI is too low to measure precisely. */
174#define RSSI_RESERVED_START 11 /* All values above and including RSSI_RESERVED_START are reserved,
175 except those defined above. */
